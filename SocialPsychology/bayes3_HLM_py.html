P_average = []
for i in range(0,6):
    P_sum = []
    for j in range(0,4):
        P_sum.append(dat.iloc[i*4+j, 4])
    P_average.append(np.mean(P_sum))

error_g=[]
for i in range(0,6):
    ave = P_average[i]
    for j in range(0,4):
        error_g.append(dat.iloc[i*4+j, 4] - ave)
dat["error_g"] = error_g

mean = np.mean(dat["group_per"])
